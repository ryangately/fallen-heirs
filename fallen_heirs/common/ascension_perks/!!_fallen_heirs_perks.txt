#################
# FALLEN HEIRS ASCENSION PERKS
#################

ap_heir_to_the_empire = {
	possible = {
		custom_tooltip = {
			fail_text = "requires_ascension_perks_2"
			num_ascension_perks > 1
		}
    }
    
	on_enabled = {
		custom_tooltip = "heir_to_the_empire_tooltip"
		hidden_effect = {
			#country_event = { id = utopia.2600 }
		}
		add_research_option = tech_dark_matter_power_core
        add_research_option = tech_dark_matter_deflector
        add_research_option = tech_dark_matter_propulsion
	}
	
	potential = {
        has_civic = civic_fallen_heir
        NOT = {
			has_ascension_perk = ap_heir_to_the_empire
		}	
	}
	
	ai_weight = {
		factor = 10
		modifier = {
			factor = 0
			NOR = {
				any_neighbor_country = { 
					is_country_type = awakened_fallen_empire
				}
				any_country = {
					is_country_type = fallen_empire
				}				
			}
		}
		modifier = {
			factor = 2
			any_country = {
				is_country_type = awakened_fallen_empire
			}
		}
	}
}

ap_heir_ascendant = {
	on_enabled = {
		custom_tooltip = "heir_ascendant_tooltip"
		hidden_effect = {
			#country_event = { id = utopia.2650 }
		}
	}
	possible = {
		custom_tooltip = {
			fail_text = "requires_ascension_perks_3"
			num_ascension_perks > 2
		}
		custom_tooltip = {
			fail_text = "requires_ap_heir_to_the_empire"
			has_ascension_perk = ap_heir_to_the_empire
		}
	}
	
	potential = {
		host_has_dlc = "Utopia"
		NOT = {
			has_ascension_perk = ap_heir_ascendant
		}	
	}
	
	ai_weight = {
		factor = 100
	}
}
